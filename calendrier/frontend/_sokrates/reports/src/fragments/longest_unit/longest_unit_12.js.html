export async function fetchGroupEvents(groupId) {
  if (!groupId) return [];

  const { data, error } = await supabase
    .from("group_events")
    .select("*")
    .eq("group_id", groupId)
    .order("start_at", { ascending: true });

  if (error) throw error;
  return data || [];
}
