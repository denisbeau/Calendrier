function generate6CharCode() {
  const letters = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
  let s = "";
  for (let i = 0; i < 6; i++)
    s += letters[Math.floor(Math.random() * letters.length)];
  return s;
}
