import React, { useState, useEffect, createContext, useContext } from 'react'; const ThemeContext = createContext({ isDark: false, toggleTheme: () => {} }); export function ThemeProvider({ children }) { const getInitial = () => { try { const stored = localStorage.getItem('theme'); if (stored === 'dark') return true; if (stored === 'light') return false; // fallback to system preference return window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches; } catch (e) { return false; } }; const [isDark, setIsDark] = useState(getInitial); useEffect(() => { try { document.body.classList.toggle('dark', isDark); document.body.classList.toggle('light', !isDark); localStorage.setItem('theme', isDark ? 'dark' : 'light'); } catch (e) { // ignore } }, [isDark]); const toggleTheme = () => setIsDark((v) => !v); return ( <ThemeContext.Provider value={{ isDark, toggleTheme }}> {children} </ThemeContext.Provider> ); } export function useTheme() { return useContext(ThemeContext); } export default ThemeContext;
